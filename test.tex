	\[	α x²+ β x+ γ= 0 \]

	\[	x_±= -\frac{β ∓ √{β² - 4αγ}}{2α}.\]

	\[	ζ(s) Γ(s)= ∫₀^∞ \frac{x^{s-1}}{eˣ-1} ⅾx,\]

	\[	Γ(s)= ∫₀^∞ x^{s-1} e^{-x} ⅾx,\]
